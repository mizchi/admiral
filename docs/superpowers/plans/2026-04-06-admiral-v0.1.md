# admiral v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** argparse の薄いラッパーとして、gunshi ライクな宣言的 CLI ビルダーを MoonBit で実装する。

**Architecture:** `OptionDef` / `CommandDef` で宣言的にコマンドを定義 → `build()` で `@argparse.Command` に変換 → `parse()` で `Context` を生成 → 型付きメソッドで値を取得。全コードは単一パッケージ `src/` に置く。

**Tech Stack:** MoonBit, moonbitlang/core/argparse

---

  {
    name,
    type_: StringOpt,
    short,
    description,
    required: if default.is_empty() { required } else { false },
    default_value: default,
  }
}

///|

  name : String,
  short~ : Char? = None,
  description~ : String = ""
) -> OptionDef {
  {
    name,
    type_: BoolOpt,
    short,
    description,
    required: false,
    default_value: None,
  }

    type_: IntOpt,
    short,

pub fn Context::get_bool(self : Context, name : String) -> Bool {
  self.flags.get(name).unwrap_or(false)
}

///|
pub fn Context::get_string(self : Context, name : String) -> String? {
  match self.values.get(name) {
    Some(arr) => if arr.length() > 0 { Some(arr[0]) } else { None }
    None => None
  }
}

///|
pub fn Context::get_string_required(self : Context, name : String) -> String raise String {
  match self.get_string(name) {
    Some(v) => v
    None => raise "missing required option: " + name
  }
}

///|
pub fn Context::get_int(self : Context, name : String) -> Int? {
  match self.get_string(name) {
    Some(s) => try { Some(@strconv.parse_int(s)) } catch { _ => None }
    None => None
  }
}

///|
pub fn Context::get_int_required(self : Context, name : String) -> Int raise String {
  match self.get_int(name) {
    Some(v) => v
    None => raise "missing or invalid required option: " + name

}

///|
pub fn Context::get_strings(self : Context, name : String) -> Array[String] {
  self.values.get(name).unwrap_or([])
}

///|
pub fn Context::get_subcommand(self : Context) -> (String, Context)? {

///|
fn build_option_to_flag(opt : OptionDef) -> @argparse.FlagArg {
  match opt.short {
    Some(c) => @argparse.FlagArg(opt.name, short=c, about=opt.description)
    None => @argparse.FlagArg(opt.name, about=opt.description)
  }
}

///|
fn build_option_to_option_arg(opt : OptionDef) -> @argparse.OptionArg {
  match (opt.short, opt.default_value, opt.required) {
    (Some(c), Some(dv), _) =>
      @argparse.OptionArg(
        opt.name, short=c, about=opt.description, default_values=[dv],
      )
    (Some(c), None, true) =>
      @argparse.OptionArg(
        opt.name, short=c, about=opt.description, required=true,
      )
    (Some(c), None, false) =>
      @argparse.OptionArg(opt.name, short=c, about=opt.description)
    (None, Some(dv), _) =>
      @argparse.OptionArg(
        opt.name, about=opt.description, default_values=[dv],
      )
    (None, None, true) =>
      @argparse.OptionArg(opt.name, about=opt.description, required=true)
    (None, None, false) =>
      @argparse.OptionArg(opt.name, about=opt.description)
  }
}

///|
pub fn build_command(cmd : CommandDef) -> @argparse.Command {
  let flags : Array[@argparse.FlagArg] = []
  let options : Array[@argparse.OptionArg] = []
  for opt in cmd.options {
    match opt.type_ {
      BoolOpt => flags.push(build_option_to_flag(opt))
      StringOpt | IntOpt => options.push(build_option_to_option_arg(opt))
    }
  }
  let subcommands : Array[@argparse.Command] = cmd.subcommands.map(
    fn(sub) { build_command(sub) },
  )
  @argparse.Command(
    cmd.name,
    about=cmd.description,
    flags=flags,
    options=options,
    subcommands=subcommands,
  )
}
```

Note: argparse の `FlagArg` / `OptionArg` コンストラクタは `StringView` を受け取る。`String` は自動変換されるはずだが、もしコンパイルエラーが出たら `.view()` を付ける。また `default_values` は `ArrayView[String]` を期待する可能性がある — エラーが出たら `[dv].view()` に修正すること。

  name~ : String,
  version~ : String = "0.0.0",
  description~ : String = "",
  options~ : Array[OptionDef] = [],
  commands~ : Array[CommandDef] = []
) -> CliApp {
  { name, version, description, root_options: options, commands }

///|
fn matches_to_context(m : @argparse.Matches) -> Context {
  let sub = match m.subcommand {
    Some((name, sub_matches)) => Some((name, matches_to_context(sub_matches)))
    None => None
  }
  { flags: m.flags, values: m.values, subcommand: sub }
}

///|
fn find_command_def(
  commands : Array[CommandDef],
  name : String
) -> CommandDef? {
  for cmd in commands {
    if cmd.name == name {
      return Some(cmd)
    }
  }
  None
}

///|
fn execute_command(
  commands : Array[CommandDef],
  ctx : Context
) -> Unit raise String {
  match ctx.subcommand {
    Some((name, sub_ctx)) =>
      match find_command_def(commands, name) {
        Some(cmd) =>
          if cmd.subcommands.length() > 0 && sub_ctx.subcommand is Some(_) {
            execute_command(cmd.subcommands, sub_ctx)
          } else {
            match cmd.run {
              Some(run_fn) => run_fn(sub_ctx)
              None =>
                raise "command '" + name + "' has no run handler"
            }
          }
        None =>
          raise "unknown command: " + name
      }
    None =>
      raise "no command specified. Use --help for usage."
  }
}

///|
pub fn CliApp::run(self : CliApp, argv : Array[String]) -> Unit raise String {
  // Build the root argparse command
  let root_cmd : CommandDef = {
    name: self.name,
    description: self.description,
    options: self.root_options,
    subcommands: self.commands,
    run: None,
  }
  let argparse_cmd = build_command(root_cmd)
  // Add version to the argparse command — rebuild with version
  let argparse_cmd_with_version = @argparse.Command(
    self.name,
    about=self.description,
    version=self.version,
    flags=argparse_cmd |> get_flags_from_built(self.root_options),
    options=argparse_cmd |> get_options_from_built(self.root_options),
    subcommands=self.commands.map(fn(c) { build_command(c) }),
  )
  let matches = argparse_cmd_with_version.parse(argv=argv)
  let ctx = matches_to_context(matches)
  execute_command(self.commands, ctx)
}

pub fn cli(
  name~ : String,
  version~ : String = "0.0.0",
  description~ : String = "",
  options~ : Array[OptionDef] = [],
  commands~ : Array[CommandDef] = []
) -> CliApp {
  { name, version, description, root_options: options, commands }
}

///|
fn matches_to_context(m : @argparse.Matches) -> Context {
  let sub = match m.subcommand {
    Some((name, sub_matches)) => Some((name, matches_to_context(sub_matches)))
    None => None
  }
  { flags: m.flags, values: m.values, subcommand: sub }
}

///|
fn find_command_def(
  commands : Array[CommandDef],
  name : String
) -> CommandDef? {
  for cmd in commands {
    if cmd.name == name {
      return Some(cmd)
    }
  }
  None
}

///|
fn execute_command(
  commands : Array[CommandDef],
  ctx : Context
) -> Unit raise String {
  match ctx.subcommand {
    Some((name, sub_ctx)) =>
      match find_command_def(commands, name) {
        Some(cmd) =>
          if cmd.subcommands.length() > 0 && sub_ctx.subcommand is Some(_) {
            execute_command(cmd.subcommands, sub_ctx)
          } else {
            match cmd.run {
              Some(run_fn) => run_fn(sub_ctx)
              None => raise "command '" + name + "' has no run handler"
            }
          }
        None => raise "unknown command: " + name
      }
    None => raise "no command specified. Use --help for usage."
  }
}

///|
fn build_root_command(app : CliApp) -> @argparse.Command {
  let root_def : CommandDef = {
    name: app.name,
    description: app.description,
    options: app.root_options,
    subcommands: app.commands,
    run: None,
  }
  build_command_with_version(root_def, app.version)
}

///|
fn build_command_with_version(
  cmd : CommandDef,
  version : String
) -> @argparse.Command {
  let flags : Array[@argparse.FlagArg] = []
  let options : Array[@argparse.OptionArg] = []
  for opt in cmd.options {
    match opt.type_ {
      BoolOpt => flags.push(build_option_to_flag(opt))
      StringOpt | IntOpt => options.push(build_option_to_option_arg(opt))
    }
  }
  let subcommands = cmd.subcommands.map(fn(sub) { build_command(sub) })
  @argparse.Command(
    cmd.name,
    about=cmd.description,
    version=version,
    flags=flags,
    options=options,
    subcommands=subcommands,
  )
}

///|
pub fn CliApp::run(
  self : CliApp,
  argv : Array[String]
) -> Unit raise String {
  let argparse_cmd = build_root_command(self)
  let matches = argparse_cmd.parse(argv=argv)
  let ctx = matches_to_context(matches)
  execute_command(self.commands, ctx)
}

# admiral

Declarative CLI builder for MoonBit, inspired by [gunshi](https://github.com/kazupon/gunshi).

A thin wrapper around `moonbitlang/core/argparse` that provides typed option helpers and a gunshi-like command definition API.

## Install

```json
{
  "deps": {
    "mizchi/admiral": "0.1.0"
  }
}
```

## Usage

```moonbit
fn main {
  let app = @admiral.cli(
    name="myapp",
    version="1.0.0",
    description="My CLI tool",
    commands=[
      @admiral.command(
        name="greet",
        description="Greet someone",
        options=[
          @admiral.string("name", short='n', description="Name to greet", required=true),
          @admiral.bool("verbose", short='v', description="Verbose output"),
          @admiral.int("count", short='c', description="Repeat count", default=1),
        ],
        run=fn(ctx) {
          let name = try { ctx.get_string_required("name") } catch { _ => return }
          let verbose = ctx.get_bool("verbose")
          let count = ctx.get_int("count").unwrap_or(1)
          for i = 0; i < count; i = i + 1 {
            if verbose {
              println("Hello, " + name + "! (greeting " + (i + 1).to_string() + ")")
            } else {
              println("Hello, " + name + "!")
            }
          }
        },
      ),
    ],
  )
  try app.run(@env.args()) catch { err => println(err) }
}
```

## API

### Option Helpers

- `string(name, short?, description?, required?, default?)` — String option
- `bool(name, short?, description?)` — Boolean flag
- `int(name, short?, description?, required?, default?)` — Integer option

### Command Definition

- `command(name, description?, options?, subcommands?, run?)` — Define a command
- `cli(name, version?, description?, options?, commands?)` — Create a CLI app

### Context Methods

- `ctx.get_bool(name) -> Bool`
- `ctx.get_string(name) -> String?`
- `ctx.get_string_required(name) -> String raise String`
- `ctx.get_int(name) -> Int?`
- `ctx.get_int_required(name) -> Int raise String`
- `ctx.get_strings(name) -> Array[String]`
- `ctx.get_subcommand() -> (String, Context)?`
